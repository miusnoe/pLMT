\subsection*{Aufgabe 1}
Wie groß ist der pH-Wert einer Salzsäure ($\text{p}K_\text{S} = -6$) mit der Konzentration $c(\text{HCl}) = 0{,}5\,\text{mol/L}$ in wässriger Lösung?

\ifshowme
\antwort{
\begin{equation*}
    \text{pH} = -\lg(0{,}5) = 0{,}3
\end{equation*}

Der pH-Wert einer $0{,}5$ molaren Salzsäure liegt bei $0{,}3$.
}
\else
\fi

\subsection*{Aufgabe 3}
Wie groß ist der pH-Wert 
\begin{itemize}
\item einer Essigsäure ($\text{p}K_\text{S} = 4{,}75$) mit der Konzentration $c = 0{,}5\,\text{mol/L}$?
\item einer Ameisensäure ($\text{p}K_\text{S} = 3{,}77$) mit der Konzentration $c = 0{,}05\,\text{mol/L}$?
\end{itemize}
\ifshowme
\antwort{
\begin{equation*}
    \text{pH} = 0{,}5 \cdot (4{,}75 - \lg(0{,}5)) = 2{,}53
\end{equation*}

Der pH-Wert einer $0{,}5$ molaren Essigsäure beträgt $2{,}53$.
}
\else
\fi
